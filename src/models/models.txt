// models/Product.ts
export interface Product {
    product_code: string;
    name: string;
    description: string;
    price: number;
    points: number;
    image: string;
}

// models/File.ts
export interface File {
    file: string;
}

// models/Consultant.ts
export interface Consultant {
    name: string;
    email: string;
    data_birth: Date;
    consultant_code: string;
    accumulated_points: number;
    entry_date: Date;
    total_sales: number;
    total_commission: number;
    status: string;
    last_activity_date: Date;
    street_address: string;
    cep: string;
    neighborhood: string;
    state: string;
    phone_number: string;
    cpf: string;
    level: string;
    digital_space_link: string;
}

// models/OrderProduct.ts
export interface OrderProduct {
    request: string;
    consultant: Consultant;
    product: Product;
    quantity: number;
    commission: number;
    total_amount: number;
    total_points: number;
}

// models/Order.ts
export interface Order {
    order_number: string;
    request: OrderProduct;
    consultant: Consultant;
    order_date: Date;
    total_amount: number;
    total_points: number;
    captacao: string;
}

// models/Cycle.ts
export interface Cycle {
    cycle_number: number;
    start_date: Date;
    end_date: Date;
}



// models/User.js
export default class User {
    constructor(id, username, email, password, role) {
        this.id = id;
        this.username = username;
        this.email = email;
        this.password = password;
        this.role = role; // Pode ser 'admin', 'distributor', 'customer', etc.
        this.referralCode = ''; // Código de referência para o marketing multinível
        this.referrerCode = ''; // Código do usuário que indicou este usuário
    }
}
